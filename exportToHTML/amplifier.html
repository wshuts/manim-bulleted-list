60       def construct(self):
61           def read_spar(path="manim_gif_data/amplifier/", parameter="s11", ext=".csv", mode="rb", delimiter=","):
62               stream = open(path + parameter + ext, mode)
63               return np.loadtxt(stream, delimiter=delimiter)

65           def scale_array_column(array, column=0, scaling_factor=1e9):
66               scaled_array = np.copy(array)
67               scaled_array[:, column] /= scaling_factor
68               return scaled_array
69   
70           def shift_array_column(array, column=1, offset=10):
71               scaled_array = np.copy(array)
72               scaled_array[:, column] += offset
73               return scaled_array

75           def create():
76               self.title = Text("Example Networks", color=YELLOW)
77               self.underline = Line(LEFT, RIGHT, color=YELLOW)
78               self.ax_amp = Axes(
79                   x_range=[13, 18, 1],
80                   y_range=[-40, 20, 10],
81                   x_length=10,
82                   y_length=6,
83                   x_axis_config={"numbers_to_include": np.arange(14, 19, 1), "label_direction": UP},
84                   y_axis_config={"numbers_to_include": np.arange(-40, 30, 10)},
85                   tips=False,
86               )
87               self.frequency_label = MathTex("\mathrm{Frequency \ (GHz)}")
88               self.magnitude_label = MathTex("\mathrm{Magnitude \ (dB)}")
89               self.circuit_amplifier = ImageMobject("amplifier.png")
90               self.port1_num = MathTex("1")
91               self.port2_num = MathTex("2")
92               self.plus = MathTex("+")
93               self.minus = MathTex("-")
94               self.plus_v = MathTex("\mathrm{+V}")
95               self.minus_v = MathTex("\mathrm{-V}")
96               self.long_right_arrow = Tex('$\Longrightarrow$')
97               self.n_squared = MathTex("n^2", "=", "4")
98               self.cap_s = MathTex(r"\mathrm{S}")
99               self.eq = MathTex("=")
100              self.s2p_matrix = Matrix([[r"\mathrm{S}_{11}", r"\mathrm{S}_{12}"],
101                                        [r"\mathrm{S}_{21}", r"\mathrm{S}_{22}"]],
102                                       left_bracket="(", right_bracket=")")
103              self.frame_box_s11 = SurroundingRectangle(self.s2p_matrix[0][0], buff=0.4)
104              self.frame_box_s11.add_updater(lambda mob: mob.move_to(self.s2p_matrix[0][0]))
105              self.frame_box_s12 = SurroundingRectangle(self.s2p_matrix[0][1], buff=0.4)
106              self.frame_box_s12.add_updater(lambda mob: mob.move_to(self.s2p_matrix[0][1]))
107              self.frame_box_s21 = SurroundingRectangle(self.s2p_matrix[0][2], buff=0.4)
108              self.frame_box_s21.add_updater(lambda mob: mob.move_to(self.s2p_matrix[0][2]))
109              self.frame_box_s22 = SurroundingRectangle(self.s2p_matrix[0][3], buff=0.4)
110              self.frame_box_s22.add_updater(lambda mob: mob.move_to(self.s2p_matrix[0][3]))
111              self.grouped_equation = VGroup(self.cap_s, self.eq, self.s2p_matrix)
112              self.s11_amp = read_spar(parameter="s11")
113              self.s21_amp = read_spar(parameter="s21")
114              self.s12_amp = read_spar(parameter="s12")
115              self.s22_amp = read_spar(parameter="s22")
116              self.s11_amp = scale_array_column(self.s11_amp)
117              self.s21_amp = scale_array_column(self.s21_amp)
118              self.s12_amp = scale_array_column(self.s12_amp)
119              self.s22_amp = scale_array_column(self.s22_amp)
120              self.s21_amp = shift_array_column(self.s21_amp, offset=-10)
121              self.s12_amp = shift_array_column(self.s12_amp)
122              self.graph_group = VGroup(self.ax_amp, self.frequency_label, self.magnitude_label)
123              self.amplifier_s11 = self.ax_amp.plot_line_graph(self.s11_amp[:, 0], self.s11_amp[:, 1],
124                                                               add_vertex_dots=False, line_color=Color(BLUE))
125              self.amplifier_s12 = self.ax_amp.plot_line_graph(self.s12_amp[:, 0], self.s12_amp[:, 1],
126                                                               add_vertex_dots=False, line_color=Color(GREEN))
127              self.amplifier_s21 = self.ax_amp.plot_line_graph(self.s21_amp[:, 0], self.s21_amp[:, 1],
128                                                               add_vertex_dots=False, line_color=Color(RED))
129              self.amplifier_s22 = self.ax_amp.plot_line_graph(self.s22_amp[:, 0], self.s22_amp[:, 1],
130                                                               add_vertex_dots=False, line_color=Color(PURPLE))
131              self.ax_amp.add(self.amplifier_s11)
132              self.ax_amp.add(self.amplifier_s21)
133              self.ax_amp.add(self.amplifier_s12)
134              self.ax_amp.add(self.amplifier_s22)
135              return

137          def create_bulleted_list(tex_strings, under_mob):
138              for tex_string in tex_strings:
139                  dot = MathTex("\\cdot")
140                  dot.scale(3).next_to(tex_string[0], LEFT * 0.4, buff=0.4)
141                  tex_string.add_to_back(dot)
142              tex_strings.arrange(DOWN, aligned_edge=LEFT, buff=0.3)
143              tex_strings.scale(1.45)
144              tex_strings.next_to(under_mob, DOWN)
145              tex_strings.set_opacity(0.5)
